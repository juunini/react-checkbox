import { Meta, Canvas, Story } from '@storybook/addon-docs';
import { useState } from 'react';

import CheckBox from '.';

<Meta title="react-checkbox" />

export const Template = (args) => {
  const [checked, setChecked] = useState(false);
  return (
    <CheckBox
      {...args}
      checked={checked}
      onChange={setChecked}
    />
  );
};
const control = {
  radio: { control: { type: 'radio' } },
  text: { control: { type: 'text' } },
  color: { control: { type: 'color' } },
  range: { control: { type: 'range', min: 0, max: 100, step: 1 } },
};
const options = {
  borderStyle: { options: ['none', 'hidden', 'solid', 'dotted', 'dashed', 'double', 'groove', 'ridge', 'inset', 'outset'] },
  checkboxPosition: { options: ['before', 'after'] },
};

<Canvas>
  <Story
    name="CheckBox"
    args={{
      marginRight: 0,
      marginLeft: 0,
      fontSize: 16,
      children: 'label',
      width: 16,
      height: 16,
      checkboxMarginRight: 10,
      checkboxMarginLeft: 10,
      borderWidth: 1,
      borderStyle: 'solid',
      borderRadius: 0,
      checkboxPosition: 'before',
    }}
    argTypes={{
      marginRight: { ...control.range },
      marginLeft: { ...control.range },
      fontSize: { ...control.range },
      width: { ...control.range },
      height: { ...control.range },
      checkboxMarginRight: { ...control.range },
      checkboxMarginLeft: { ...control.range },
      borderWidth: { ...control.range },
      borderStyle: { ...control.radio, ...options.borderStyle },
      borderColor: { ...control.color },
      borderRadius: { ...control.range },
      checkboxPosition: { ...control.radio, ...options.checkboxPosition },
      checkedBackgroundColor: { ...control.color },
      checkedContent: { ...control.text },
      checkedImage: { ...control.text },
      checkedImageSize: { ...control.text },
    }}
  >
    {Template.bind({})}
  </Story>
</Canvas>
