import { Meta, Story, Canvas } from '@storybook/addon-docs';
import { useState } from 'react';

import Switch from './Switch';

<Meta
  title="Switch"
  component={Switch}
/>

export const SwitchTemplate = (args) => {
  const [checked1, setChecked1] = useState(false);
  const [checked2, setChecked2] = useState(false);
  const [checked3, setChecked3] = useState(false);
  const [checked4, setChecked4] = useState(false);
  return (
    <>
      <Switch {...args} type="checkbox" checked={checked1} onChange={setChecked1}>label</Switch>
      <Switch {...args} type="checkbox" checked={checked2} onChange={setChecked2}>label</Switch>
      <Switch {...args} type="checkbox" checked={checked3} onChange={setChecked3}>label</Switch>
      <Switch {...args} type="checkbox" checked={checked4} onChange={setChecked4}>label</Switch>
    </>
  );
};

export const RadioTemplate = (args) => {
  const [value, setValue] = useState(false);
  return (
    <>
      <Switch {...args} type="radio" name="radio" value="1" checked={value === '1'} onChange={setValue}>label1</Switch>
      <Switch {...args} type="radio" name="radio" value="2" checked={value === '2'} onChange={setValue}>label2</Switch>
      <Switch {...args} type="radio" name="radio" value="3" checked={value === '3'} onChange={setValue}>label3</Switch>
      <Switch {...args} type="radio" name="radio" value="4" checked={value === '4'} onChange={setValue}>label4</Switch>
    </>
  );
};

<Canvas>
  <Story
    name="checkbox"
    args={{
      labelStyle: {
        margin: '0 10px',
      },
      switchStyle: {
        marginRight: '5px',
      },
    }}
  >
    {SwitchTemplate.bind({})}
  </Story>
</Canvas>

<Canvas>

      },
    }}
  >
